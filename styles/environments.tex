\tcbuselibrary{skins, breakable, theorems}

% استایل ۱: باکس با هدر روبان شکل (Ribbon Box) برای قضیه، تعریف و ...
\makeatletter
\tcbset{
	ribbon_box/.style={
		enhanced,
		breakable,
		before=\par\pagebreak[0]\needspace{4\baselineskip}\noindent,
		colback=white,
		attach boxed title to top right={yshift*=-\tcboxedtitleheight},
		fonttitle=\bfseries\setRTL,
		before upper=\setRTL,
		boxed title size=title,
		boxed title style={
			sharp corners,
			rounded corners=northeast,
			colback=tcbcolframe,
			boxrule=0pt,
		},
		underlay boxed title={
			\path[fill=tcbcolframe] (title.south east)--(title.south west)
			to[out=180, in=0] ([xshift=-5mm]title.west)--
			(title.center-|frame.west)
			[rounded corners=\kvtcb@arc] |-
			(frame.north) -| cycle;
		},
	}
}
\makeatother

% استایل ۲: باکس مینیمال با خط کناری (Side Box) برای مثال، حل، نکته و ...
\tcbset{
	side_box/.style={
		enhanced,
		breakable,
		before=\par\pagebreak[0]\needspace{3\baselineskip}\noindent,
		frame hidden,
		boxrule = 0sp,
		borderline east = {2pt}{0pt}{#1}, % خط کناری در سمت راست
		sharp corners,
		detach title,
		before upper = \setRTL\tcbtitle\par\smallskip,
		coltitle = #1,
		fonttitle = \bfseries\setRTL,
		separator sign none,
		segmentation style={solid, #1},
	}
}

% شمارنده مشترک برای همه محیط‌های ریاضی که وابسته به فصل یا بخش است
\makeatletter
\@ifundefined{c@chapter}{
    \newcounter{mathbox}[section]
    \renewcommand{\themathbox}{\thesection.\arabic{mathbox}}
}{
    \newcounter{mathbox}[chapter]
    \renewcommand{\themathbox}{\thechapter.\arabic{mathbox}}
}
\makeatother
