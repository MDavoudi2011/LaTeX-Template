% این فایل شامل ۳ استایل زیبا و متفاوت برای جداول با پشتیبانی کامل از راست‌چین (RTL) است.

% ---------------------------------------------------------
% راهنمای استفاده:
% 1. عرض و چینش ستون‌ها (پارامتر سوم):
%    c (وسط‌چین)، r (راست‌چین)، l (چپ‌چین)، p{3cm} (عرض ثابت)
%    برای اینکه جدول هم‌عرض صفحه شود، می‌توانید به جای tabular از tabularx استفاده کنید
%    و از ستون نوع X استفاده کنید. اما در اینجا برای سادگی و هماهنگی از tabular استفاده شده است.
%
% 2. هدر جداول (ردیف اول):
%    در جداول رنگی (مدرن و فانتزی) کلمات هدر را داخل \THead{} قرار دهید تا سفید و توپر شوند.
%    مثال: \THead{نام} & \THead{سن}
%
% 3. خطوط جدول (مینیمال):
%    در جدول مینیمال فقط برای جدا کردن هدر از محتوا، در انتهای سطر اول از \midrule استفاده کنید.
%
% 4. خطوط جدول (مدرن):
%    در این جدول نیازی به هیچ خطی (\hline یا \midrule) نیست! فقط سطرها را بنویسید.
%
% 5. خطوط جدول (فانتزی):
%    برای خط‌کشی کامل، در تعریف ستون‌ها از | استفاده کنید (مثل {|c|c|c|})
%    و بین تمام سطرها \hline بگذارید. رنگ خطوط به طور خودکار همرنگ کادر می‌شود.
% ---------------------------------------------------------

% دستور کمکی برای هدر جداول رنگی (رنگ متن سفید و توپر)
\newcommand{\THead}[1]{\textcolor{white}{\textbf{#1}}}

% 1. جدول مینیمال (Minimal Table)
\NewDocumentEnvironment{minimaltable}{ O{H} m m +b }{%
    \begin{table}[#1]
        \centering
        \caption{#2}
        \renewcommand{\arraystretch}{1.3} % فاصله مناسب سطرها
        \begin{tabular}{#3}
            \toprule
            #4
            \bottomrule
        \end{tabular}
    \end{table}
}{}

% 2. جدول مدرن (Modern Table)
\NewDocumentEnvironment{moderntable}{ O{H} m m +b }{%
    \begin{table}[#1]
        \centering
        \caption{#2}
        \renewcommand{\arraystretch}{1.3}
        \rowcolors{2}{ThemeSecondary!10}{white}
        \begin{tabular}{#3}
            \rowcolor{ThemeSecondary}
            #4
        \end{tabular}
    \end{table}
}{}
