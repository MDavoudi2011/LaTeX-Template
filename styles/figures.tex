% تعریف استایل سایه‌ها برای جلوگیری از خطای پردازش در forest
\tikzset{
  tree shadow/.style={
    drop shadow={opacity=0.15, shadow xshift=1.5pt, shadow yshift=-1.5pt}
  }
}

\forestset{
  default tree/.style={
    for tree={
      draw=ThemePrimary,
      thick,
      top color=ThemeSecondary!20,
      bottom color=ThemeSecondary!10,
      text=black,
      rounded corners=6pt,
      edge={draw=ThemePrimary, thick, -latex},
      tree shadow,
      inner sep=8pt,
      font=\small\bfseries,
      align=center,
      execute at begin node={\setRTL},
      s sep=1cm,
      l sep=1.2cm,
      parent anchor=south,
      child anchor=north
    }
  }
}

% ترفند باز کردن محتوای \BODY قبل از اجرای forest
\def\RenderTree#1{\begin{forest} default tree #1 \end{forest}}

\NewEnviron{tree}{%
  \begin{center}
  \expandafter\RenderTree\expandafter{\BODY}
  \end{center}
}
