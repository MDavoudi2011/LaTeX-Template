% شمارنده مشترک برای همه محیط‌های ریاضی که وابسته به فصل یا بخش است
\makeatletter
\@ifundefined{c@chapter}{
    \newcounter{mathbox}[section]
    \renewcommand{\themathbox}{\thesection.\arabic{mathbox}}
}{
    \newcounter{mathbox}[chapter]
    \renewcommand{\themathbox}{\thechapter.\arabic{mathbox}}
}
\makeatother

% ------------------------------------------------------------------------------
% محیط‌های شماره‌دار
% ------------------------------------------------------------------------------

% 1. قضیه
\NewTColorBox[use counter=mathbox]{theorem}{ o }{
    ribbon_box,
    colframe=BoxBlueFrame,
    label type=theorem,
    title={\faLightbulb \quad قضیه \themathbox\IfValueT{#1}{. \textnormal{#1}}}
}

% 2. لم
\NewTColorBox[use counter=mathbox]{lemma}{ o }{
    ribbon_box,
    colframe=BoxBlueFrame,
    label type=lemma,
    title={\faLink \quad لم \themathbox\IfValueT{#1}{. \textnormal{#1}}}
}

% 3. تعریف
\NewTColorBox[use counter=mathbox]{definition}{ o }{
    ribbon_box,
    colframe=BoxGreenFrame,
    label type=definition,
    title={\faBook \quad تعریف \themathbox\IfValueT{#1}{. \textnormal{#1}}}
}

% 4. مثال
\NewTColorBox[use counter=mathbox]{example}{ o }{
    ribbon_box,
    colframe=BoxTealFrame,
    label type=example,
    title={\faEdit \quad مثال \themathbox\IfValueT{#1}{. \textnormal{#1}}}
}

% 5. سوال
\NewDocumentEnvironment{question}{}{%
    \par\pagebreak[0]\needspace{3\baselineskip}\vspace{0.25\baselineskip}\noindent
    \refstepcounter{mathbox}%
    {\bfseries\color{ThemePrimary}\faQuestionCircle \quad سوال \themathbox.} \quad\ignorespaces
}{%
    \par\vspace{0.25\baselineskip}
}
