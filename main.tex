\chapter{مقدمه و شروع به کار}
سلام! به قالب مدرن جزوه‌نویسی خوش آمدید. 
هدف از طراحی این ساختار، ایجاد بستری ماژولار، تمیز و بسیار حرفه‌ای است تا نوشتن متون علمی به سادگی \E{Markdown} و با خروجی استاندارد \E{LaTeX} انجام شود.

\section{تست فرمول‌ها و اعداد}
در این قالب از پکیج‌های پیشرفته ریاضی استفاده شده است. به عنوان مثال:
\begin{equation}
    \int_{-\infty}^{+\infty} e^{-x^2} dx = \sqrt{\pi}
\end{equation}
همانطور که می‌بینید، تنظیمات به گونه‌ای است که اعدادی مثل ۱، ۲ و ۳ در متن فارسی به درستی و با فونت اختصاص‌یافته نمایش داده می‌شوند.

\section{تست محیط‌های سفارشی}

\begin{definition}[هوش مصنوعی]
هوش مصنوعی (AI) به شاخه‌ای از علوم رایانه گفته می‌شود که هدف آن تولید ماشین‌های هوشمند است.
\end{definition}

\begin{theorem}[قضیه اساسی حساب دیفرانسیل]
فرض کنید $f$ تابعی پیوسته در بازه $[a, b]$ باشد. آنگاه...
\end{theorem}
\begin{proof}
این یک محیط اثبات بسیار تمیز است که در انتهای آن یک مربع توخالی قرار می‌گیرد.
\end{proof}

\begin{example}
برای مثال، تابع $f(x) = x^2$ را در نظر بگیرید.
\end{example}
\begin{question}
تابع $f(x) = x^3 + 2x$ داده شده است. مشتق این تابع را به دست آورید و مقدار آن را در $x=1$ محاسبه کنید.
\end{question}
\begin{solution}
مشتق تابع برابر با $f'(x) = 3x^2 + 2$ خواهد بود. بنابراین مقدار آن در $x=1$ برابر است با ۵.
\end{solution}

\begin{note}
شما می‌توانید به راحتی و بدون درگیر شدن با استایل‌های پیچیده، این باکس‌ها را ایجاد کنید.
\end{note}

\end{warning}
