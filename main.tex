\chapter{مقدمه و ارجاع‌دهی}
سلام! به قالب مدرن جزوه‌نویسی خوش آمدید. 
در این قالب سیستم ارجاع‌دهی هوشمند (\E{cleveref}) پیاده‌سازی شده است. به جای نوشتن دستی کلماتی مثل «بخش»، «شکل» یا «قضیه»، فقط کافیست از دستور \E{\textbackslash cref} استفاده کنید.

به عنوان مثال، به \cref{sec:math}، \cref{thm:fundamental}، \cref{fig:sample} و \cref{tab:sample} توجه کنید! تمام این ارجاعات به صورت خودکار کلمه مناسب را در کنار شماره قرار می‌دهند و با کلیک روی تمام متن آن به بخش مربوطه منتقل می‌شوید.

\section{تست فرمول‌ها و اعداد}
\label{sec:math}
در این قالب از پکیج‌های پیشرفته ریاضی استفاده شده است. به عنوان مثال:
\begin{equation}
    \label{eq:integral}
    \int_{-\infty}^{+\infty} e^{-x^2} dx = \sqrt{\pi}
\end{equation}
همانطور که در \cref{eq:integral} می‌بینید، تنظیمات به گونه‌ای است که اعداد در متن فارسی به درستی نمایش داده می‌شوند.

\section{تست محیط‌های سفارشی}

\begin{definition}[هوش مصنوعی]
\label{def:ai}
هوش مصنوعی (AI) به شاخه‌ای از علوم رایانه گفته می‌شود که هدف آن تولید ماشین‌های هوشمند است.
\end{definition}

\begin{theorem}[قضیه اساسی حساب دیفرانسیل]
\label{thm:fundamental}
فرض کنید $f$ تابعی پیوسته در بازه $[a, b]$ باشد. آنگاه...
\end{theorem}
\begin{proof}
این یک محیط اثبات بسیار تمیز است که در انتهای آن یک مربع توخالی قرار می‌گیرد.
\end{proof}

\begin{example}
\label{ex:sample}
برای مثال، تابع $f(x) = x^2$ را در نظر بگیرید.
همچنین همانطور که در \cref{def:ai} و \cref{ex:sample} مشخص است، شماره‌گذاری‌ها کاملاً یکپارچه هستند.
\end{example}
\begin{question}
تابع $f(x) = x^3 + 2x$ داده شده است. مشتق این تابع را به دست آورید و مقدار آن را در $x=1$ محاسبه کنید.
\end{question}
\begin{solution}
مشتق تابع برابر با $f'(x) = 3x^2 + 2$ خواهد بود. بنابراین مقدار آن در $x=1$ برابر است با ۵.
\end{solution}

\begin{note}
شما می‌توانید به راحتی و بدون درگیر شدن با استایل‌های پیچیده، این باکس‌ها را ایجاد کنید.
\end{note}

\end{warning}
